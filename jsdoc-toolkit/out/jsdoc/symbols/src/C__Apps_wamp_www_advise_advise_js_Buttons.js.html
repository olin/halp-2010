  7 /**
  8 Callback for a click on one of the main tabs
  9 @param obj the tab that was clicked
 10 */
 11 Buttons.maintabClick = function(/**Element*/ obj) {
 12 	var jqdiv = $(obj);
 13 	
 14 	//click on another tab
 15 	if (!jqdiv.hasClass("down")) {
 16 		//unhighlight the previous tab
 17 		$(".maintab.down").removeClass("down");
 18 		jqdiv.addClass("down");
 19 		
 20 		//hide all tabs
 21 		$(".ui-layout-center").children().each(function(i, el) {
 22 			$(el).css("display","none");
 23 		});
 24 		
 25 		//show the relevant tab
 26 		var name = jqdiv.attr("id").split("_")[1];
 27 		$("#cnt_" + name).css({
 28 			"display":"block"
 29 		});
 30 		
 31 		ContentPane.switchTo(name);
 32 	}
 33 };

 35 /**
 36 Callback for "print" button on main UI
 37 */
 38 Buttons.print = function() {
 39 	$.post("db/print.php", 
 40 		{
 41 			"data":JSON.stringify(Store.printserialize()),
 42 			"credits":JSON.stringify(Store.printcompletion()),
 43 			"username":curuser.getName()
 44 		},
 45 		function(text) {
 46 			//splat the returned text into the dialog
 47 			var del = $("#dialog");
 48 			del.attr("title","Print Status");
 49 			del.empty();
 50 			var dtxt = $("<div style='font-size:110%;'></div>")
 51 				.html(text);
 52 			var expltxt = $("<div></div>")
 53 				.text("Click the link above to download or print your plan.");
 54 			del.append(dtxt);
 55 			del.append(expltxt);
 56 			
 57 			//show the dialog
 58 			del.dialog({ 
 59 				buttons: { 
 60 					"Ok": function() { 
 61 						$(this).dialog("close"); 
 62 					} 
 63 				},
 64 				autoOpen: false
 65 			});
 66 			del.dialog("open");
 67 		}, "text");
 68 };

 73 Buttons.save = function() {

 81 				"major":Schedule.getMajor()
 82 			},
 83 			function(data) {
 84 				//ignore data sent back from the server
 85 			}, "text");
 86 		
 87 		//upon save, reset the modification flag
 88 		Store.setchanged(false);
 89 		currentlySaving = false;
 90 	});
 91 };

104 /**
105 Perform a search using updatesearch. Use a delay and ensure that multiple calls do not occur
106 */
107 Buttons.dosearch = function() {
108 	if (!Buttons.searching) {
109 		$("#searchimg").attr("src","img/progress.gif");
110 		setTimeout(Buttons.updatesearch, 300);
111 	}
112 	Buttons.searching = true;
113 };

115 /**
116 Update the course search area
117 */
118 Buttons.updatesearch = function() {
119 	//split by a comma delimiter in the search
120 	var texts = $("#search").attr("value").split(",");
121 	$("#clinner").children().each(function(i) {
122 		var course = $(this).data("course");
123 		var strings = [course["Code"], course["Nickname"], course["Title"], course["Instructors"]];
124 		//categories in the UI
125 		var cats = [true, true, $("#search_title").attr("checked"), $("#search_faculty").attr("checked")];
126 		
127 		//match to courses
128 		var match = false;
129 		for (var x = 0; x < strings.length; ++x) {
130 			if (cats[x]) {
131 				for (var y = 0; y < texts.length; ++y) {
132 					if (strings[x].toLowerCase().indexOf(texts[y].toLowerCase()) > -1) {
133 						match = true;
134 					}
135 				}
136 			}
137 		}
138 		
139 		$(this).css("display", (match || texts == [""]) ? "inline" : "none");
140 	});
141 	Buttons.searching = false;
142 	$("#searchimg").attr("src","img/find.png");
143 };
