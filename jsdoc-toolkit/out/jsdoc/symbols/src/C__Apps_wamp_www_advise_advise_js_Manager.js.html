 14 @param fn continuation function to use

 16 Manager.init = function(/**Function*/ fn) {
 17 	if (Manager.initialized) {
 18 		fn();
 19 	}

 21 	$.getJSON("db/req.php?q=getcoursesources", {rstr: Utils.rstr()},
 22 	 function(data) {

 24 		Manager.sources = [];
 25 		for (var x = 0; x < data.length; ++x) {

 35 /**
 36 Insert the list of sources into the given UI element
 37 @param elm the select box to use
 38 */
 39 Manager.createSourceList = function(/**jquery*/ elm) {
 40 	//insert the source options
 41 	for (var x = 0; x < this.sources.length; ++x) {
 42 		var source = this.sources[x];
 43 		elm.append($("<option></option>")
 44 			.attr("value",source)
 45 			.html(source)
 46 		);
 47 	}
 48 	//create the list
 49 	this.selectSource($("#manage_sourcesel").get(0));
 50 };

 88 /**
 89 Loads the course list, as given by the current selection option
 90 @param elm the select box to use as the selection source
 91 */
 92 Manager.selectSource = function(/**Element*/ elm) {
 93 	var sourcev = $(elm).attr("value");
 94 	Manager.cursource = sourcev;
 95 
 96 	var insertlst = $("#manage_courseinsert");
 97 	
 98 	//clear and display the loading information
 99 	insertlst.empty();
100 	var intr;
101 	insertlst.append(
102 		intr = $("<tr></tr>")
103 	);
104 	intr.append(
105 		$("<td></td>")
106 			.attr("colspan", 2)
107 			.text("Loading...")	
108 	);
109 	
110 	$.getJSON("db/req.php?q=getcourseeditlist", {source: sourcev}, function(data) {
111 		var insertlst = $("#manage_courseinsert");
112 		var intr;
113 		
114 		insertlst.empty();
115 		
116 		for (var x = 0; x < data.length; ++x) {
117 			var course = data[x];
118 			//do the insert of the course list
119 			insertlst.append(
120 				intr = $("<tr onclick='Manager.selectCourse(this)'></tr>")
121 					.data("Code", course["Code"])
122 			);
123 			intr.append(
124 				$("<td></td>").html(course["Code"].replace(/\ /ig, " "))
125 			);
126 			intr.append(
127 				$("<td></td>").html(course["Nickname"])
128 			);
129 		}
130 	});
131 
132 	//$("#manage_courseinsert")
133 };

135 /**
136 Callback for selecting a course from the edit list on the left pane of the management interface
137 @param elm the clicked table row element
138 */
139 Manager.selectCourse = function(/**Element*/ elm) {
140 	var qel = $(elm);
141 	
142 	//dehighlight any previously selected course
143 	$(".courseeditlist tbody tr.selected").each(function(i, el) {
144 		$(el).removeClass("selected");
145 	});
146 	qel.addClass("selected");
147 	
148 	//load the course editor interface
149 	this.CourseEditor.init();
150 	
151 	//load the revisions list
152 	$.getJSON("db/req.php?q=getrevisions", 
153 		{source: Manager.getSource(), code: qel.data("Code"), rstr: Utils.rstr()}, 
154 		function(data) {
155 			var revs = data;
156 			
157 			var intr;
158 			var insertlst = $("#manage_revisionsinsert");
159 			insertlst.empty();
160 			
161 			//insert each revision into the table
162 			for (var x = 0; x < revs.length; ++x) {
163 				var rev = revs[x];
164 				insertlst.append(
165 					intr = $("<tr onclick='Manager.selectRevision(this)'></tr>")
166 						.data("Code", qel.data("Code"))
167 						.data("SaveTime", rev["SaveTime"])
168 						.data("Current", (x == 0))
169 					);
170 					
171 					var dat = new Date();
172 					dat.setTime(rev["SaveTime"]);
173 					intr.append(
174 						$("<td></td>").html(dat.toDateString())
175 					);
176 					intr.append(
177 						$("<td></td>").html(rev["SaveUser"])
178 					);
179 					
180 				if (x == 0) {
181 					Manager.selectRevision(intr);
182 				}
183 			}
184 		}
185 	);
186 };

188 /**
189 Callback for selecting a revision from the revisions list
190 @param elm the clicked table row element
191 */
192 Manager.selectRevision = function(/**Element*/ elm) {
193 	var qel = $(elm);
194 
195 	//dehighlight any previously selected revision
196 	$(".revisionslist tbody tr.selected").each(function(i, el) {
197 		$(el).removeClass("selected");
198 	});
199 	qel.addClass("selected");
200 
201 	if (qel.data("Current")) {
202 		//if it's current, load the associated course
203 		$.getJSON("db/req.php?q=getcourse", 
204 			{source: Manager.getSource(), code: qel.data("Code"), rstr: Utils.rstr()}, 
205 			function(data) {
206 				var course = data[0];
207 				Manager.CourseEditor.setFields(course);
208 				Manager.CourseEditor.setEditability(Manager.CourseEditor.EDIT_EXISTING);
209 				Manager.CourseEditor.setAction(
210 					Manager.CourseEditor.ACTION_SAVE_REVISION,
211 					{
212 						Source: Manager.getSource(),
213 						Code: qel.data("Code")
214 					}
215 					);
216 			}
217 		);	
218 	} else {
219 		//otherwise, load the revision
220 		$.getJSON("db/req.php?q=getrevision", 
221 			{source: Manager.getSource(), code: qel.data("Code"), savetime: qel.data("SaveTime"), rstr: Utils.rstr()}, 
222 			function(data) {
223 				var course = data[0];
224 				Manager.CourseEditor.setFields(course);
225 				Manager.CourseEditor.setEditability(Manager.CourseEditor.EDIT_NONE);
226 				Manager.CourseEditor.setAction(
227 					Manager.CourseEditor.ACTION_REVERT_TO,
228 					{
229 						Source: Manager.getSource(),
230 						Code: qel.data("Code"),
231 						SaveTime: qel.data("SaveTime")
232 					}
233 					);
234 			}
235 		);	
236 	}
237 };

309 /**
310 Set the course editor to have the given edit possibilities
311 */
312 Manager.CourseEditor.setEditability = function(/**int*/ status) {
313 	var edit = [];
314 	switch (status) {
315 		case Manager.CourseEditor.EDIT_ALL:
316 			edit = [true, true, true, true, true, true, true, true, true, true];
317 			break;
318 		case Manager.CourseEditor.EDIT_EXISTING:
319 			edit = [false, false, true, true, true, true, true, true, true, true];
320 			break;
321 		case Manager.CourseEditor.EDIT_NONE:
322 			edit = [false, false, false, false, false, false, false, false, false, false];
323 			break;
324 		default:
325 			error("Invalid CourseEditor edit state: " + status);
326 			break;
327 	}
328 	
329 	for (var x = 0; x < this.editors.length; ++x) {
330 		this.editors[x].setEditable(edit[x]);
331 	}
332 };

334 /**
335 Set the fields of the course editor area to the content of the given course
336 */
337 Manager.CourseEditor.setFields = function(/**Course*/ course) {
338 	for (var x = 0; x < this.editors.length; ++x) {
339 		//lookup the field in the course
340 		var editor = this.editors[x];
341 		editor.setText(course[editor.name]);
342 	}
343 };

354 /**
355 Set the action of the editor button
356 @param action one of ACTION_SAVE_REVISION or ACTION_REVERT_TO
357 @param properties the necessary database parameters for the action
358 */
359 Manager.CourseEditor.setAction = function(/**int*/ action, /**dict*/ properties) {
360 	Manager.CourseEditor.action = [action, properties];
361 	
362 	switch (action) {
363 		case Manager.CourseEditor.ACTION_SAVE_REVISION:
364 			this.button.text("Save New Revision");
365 			break;
366 		case Manager.CourseEditor.ACTION_REVERT_TO:
367 			this.button.text("Revert To This Revision");
368 			break;
369 		default:
370 			error("Invalid CourseEditor button action: " + action);
371 			break;
372 	}
373 };

375 Manager.CourseEditor.doAction = function() {
376 	var action = Manager.CourseEditor.action[0];
377 	var props = Manager.CourseEditor.action[1];
378 	var editors = Manager.CourseEditor.editors;
379 	
380 	switch (action) {
381 		case Manager.CourseEditor.ACTION_SAVE_REVISION:
382 			var dct = {};
383 			//load up k/v pairs
384 			for (var x = 0; x < editors.length; ++x) {
385 				dct[editors[x].name] = editors[x].getText();
386 			}
387 			dct["Source"] = props["Source"];
388 			dct["SaveUser"] = curuser.username;
389 			dct["SaveTime"] = (new Date()).getTime();
390 			
391 			$.post("db/req.php?q=makerevision", dct, function(text) {
392 				//reload the course
393 				Manager.selectCourse($(".courseeditlist tbody tr.selected").get(0));
394 			}, "text");
395 			break;
396 		case Manager.CourseEditor.ACTION_REVERT_TO:
397 			$.prompt(
398 				"<b>Are you sure you would like to revert to this revision?</b><br>" +
399 				"This action cannot be undone, and will affect all users. You will be listed as the revision author.",
400 				{ 
401 					buttons: { Yes: true, No: false },
402 					callback: function(v, m) {
403 						//the value is only true if the user has clicked yes
404 						if (v) {
405 							//should have Source, Code, and SaveTime
406 							var dct = props;
407 							dct["NewUser"] = curuser.username;
408 
409 							$.post("db/req.php?q=revertto", dct, function(text) {
410 								//reload the course	
411 								Manager.selectCourse($(".courseeditlist tbody tr.selected").get(0));
412 							});
413 						}
414 					}
415 				}
416 				);
417 			break;
418 	}
419 };

421 /**
422 @class Creates an editor for a piece of data
423 */
424 Manager.DataEditor = function(/**String*/ name, /**int*/ type, /**Boolean*/ editable) {
425 	this.name = name;
426 	
427 	this.mainbox = $("<div>")
428 		.addClass("dataeditor");
429 		
430 	this.descbox = $("<div>")
431 		.text(name)
432 		.addClass("datadesc")
433 		.appendTo(this.mainbox);
434 		
435 	this.editimg = $("<img>")
436 		.css("float","left")
437 		.attr("alt","editable")
438 		.appendTo(this.mainbox);
439 		
440 	switch (type) {
441 		case Manager.DataEditor.SINGLETEXT:
442 			this.field = $("<input type='text'>");
443 			break;
444 		case Manager.DataEditor.MULTITEXT:
445 			this.field = $("<textarea></textarea>");
446 			break;
447 		default:
448 			this.field = $("<div>")
449 				.text("Invalid data editor type: " + type);
450 	}
451 
452 	this.setEditable(editable);
453 	
454 	this.field.appendTo(this.mainbox);
455 };

457 /**
458 Make the editor area enabled or disabled
459 @param editable should it be enabled or disabled?
460 */
461 Manager.DataEditor.prototype.setEditable = function(/**Boolean*/ editable) {
462 	if (editable) {
463 		this.field.removeAttr("readonly");
464 		this.editimg.attr("src", "img/application_form_edit.png");
465 	} else {
466 		this.field.attr("readonly","readonly");
467 		this.editimg.attr("src", "img/application_form_edit_cross.png");
468 	}
469 };
