  7 /**
  8 Locate a semester tick where a drop was occuring. Arguments: each point has x,y coordinates.
  9 @param lowercorner the very top corner of all divs
 10 @param firstcorner a point one "tick" away in each direction from the lowercorner
 11 @param pos the location where the element was dropped
 12 @return [semester, credit], where semester corresponds to a semester index and credit corresponds to the credit tick.
 13  -1 for either if it isn't in bounds.
 14 */
 15 Box.identifyDropLoc = function(/**Array*/ lowercorner, /**Array*/ firstcorner, /**Array*/ pos) {
 16 	var xr = pos[0] - lowercorner[0];
 17 	var yr = pos[1] - lowercorner[1];
 18 	
 19 	var xsp = firstcorner[0] - lowercorner[0];
 20 	var ysp = firstcorner[1] - lowercorner[1];
 21 	
 22 	var sem = Math.floor(yr/ysp);
 23 	if (sem < 0 || sem >= Ticks.NUMSEMS) {
 24 		sem = -1;
 25 	}
 26 	var tick = Math.floor(xr/xsp);
 27 	if (tick < 0 || tick >= Ticks.MAXCREDITS) {
 28 		tick = -1;
 29 	}
 30 	
 31 	return [sem, tick];
 32 }

 34 /**
 35 Make the given jQuery object a draggable course/metacourse
 36 @param jqdiv the box to make draggable
 37 */
 38 Box.makeDraggable = function(jqdiv) {
 39 	jqdiv.draggable({
 40 			appendTo: document.body,
 41 			helper: "clone",
 42 			zIndex: 500,
 43 			opacity: 0.5,
 44 			delay: 100,
 45 			cursor: "move",
 46 			start: function(e, ui) {
 47 				var src = Box.srcFromDrag(e);
 48 				this.ismeta = $(src).hasClass("metac");
 49 				var totalcred = $(src).data("numcredits");
 50 				var course = $(src).data("course");
 51 				
 52 				//hide the dragging course if it was not from "all courses"
 53 				if (src.parentNode.id != "clinner") {
 54 					$(src).css("display","none");
 55 				}
 56 				
 57 				//highlight the accepting blocks
 58 				for (var x = 0; x < Ticks.NUMSEMS; ++x) {
 59 					var semcred = Schedule.getSemesterCredits(x);
 60 					if (semcred + totalcred <= Ticks.MAXCREDITS) {
 61 						Ticks.highlightTicks(x, semcred, semcred + totalcred);
 62 					}
 63 				}
 64 				
 65 				//obtain measurements for where the semesters actually are
 66 				this.lowercorner = $("#sem-0-0").offset();
 67 				this.lowercorner = [this.lowercorner.left, this.lowercorner.top];
 68 				this.firstcorner = $("#sem-1-1").offset();
 69 				this.firstcorner = [this.firstcorner.left, this.firstcorner.top];
 70 				this.parentsem = src.parentNode;
 71 				var psemno = parseInt(src.parentNode.id.substring(3));
 72 				
 73 				if (!this.ismeta) {
 74 					var accepting = [];
 75 					//highlight all accepting metaboxes and store them
 76 					$(".metac").each(function(i, el) {
 77 						var qel = $(el);
 78 						var acceptstate = qel.data("rule").accept(course["Code"]);
 79 						if (acceptstate > 0) {
 80 							accepting.push([acceptstate, el]);
 81 							var highlight = Rules.acceptColorForCode(acceptstate);
 82 							qel.css("border","2px solid " + highlight);
 83 							$(qel.children()[0]).css("background-color",highlight);
 84 							
 85 							//if the accepting metabox previously held this course,
 86 							//clear the box's accept state
 87 							if (psemno == parseInt(el.parentNode.id.substring(3))) {
 88 								qel.data("hascourse", false);
 89 							}
 90 						}
 91 					});
 92 					this.accepting = accepting;
 93 				}
 94 				
 95 				this.dragitem = src;
 96 			},
 97 			drag: function(e, ui) {
 98 				//var smt = Box.identifyDropLoc(this.lowercorner, this.firstcorner, [e.clientX, e.clientY]);
 99 			},
100 			stop: function(e, ui) {
101 				var src = this.dragitem;
102 				if (src == null) { return; }
103 				//was the dragged course from the schedule?
104 				var fromschedule = $(src.parentNode).hasClass("semblock");
105 				var totalcred = $(src).data("numcredits");
106 				var newcourse = $(src).data("course");
107 				var coords = [e.originalEvent.clientX, e.originalEvent.clientY];
108 				var smt = Box.identifyDropLoc(this.lowercorner, this.firstcorner, coords);
109 				var successfuldrop = smt[0] != -1 && smt[1] != -1;
110 			
111 				//unhighlight all highlighted tick marks
112 				for (var x = 0; x < Ticks.NUMSEMS; ++x) {
113 					var semcred = Schedule.getSemesterCredits(x);
114 					Ticks.delightTicks(x, Math.max(0, semcred - totalcred - 1), 
115 									Math.min(Ticks.MAXCREDITS, semcred + totalcred + 1));
116 				}
117 
118 				if (!this.ismeta) {
119 					var oneremoved = false;
120 					
121 					//unhighlight all accepting metaboxes
122 					for (var x = 0; x < this.accepting.length; ++x) {
123 						var qel = $(this.accepting[x][1]);
124 						var normal = Utils.colorstr(qel.data("course").Color);
125 						qel.css("border","2px solid " + normal);
126 						$(qel.children()[0]).css("background-color",normal);
127 					}
128 				}
129 
130 				//successful drop
131 				if (successfuldrop) {
132 					var semcred = Schedule.getSemesterCredits(smt[0]);
133 
134 					var dropped = $(src).data("clone")();
135 					dropped.css({
136 						"position":"absolute",
137 						"left":semcred*Ticks.credw,
138 						"height":SEMHEIGHT,
139 						"top":(this.ismeta) ? "-15px" : "0px"
140 						/*"z-index":500*/
141 					}).appendTo("#sem" + smt[0]);
142 					
143 					if (!this.ismeta) {
144 						Tips.updateAlerts(dropped);
145 					}
146 				} 
147 				
148 				//remove the old course it was already on the schedule.
149 				if (fromschedule) {
150 					//clear credits before repacking divs; this course will be removed soon anyway.
151 					$(src).data("numcredits",0);
152 					Schedule.repackDivs(src.parentNode.id.substring(3));
153 					src.parentNode.removeChild(src);
154 				}
155 				
156 				Schedule.repackDivs(smt[0]);
157 				
158 				Store.setchanged(true);
159 				
160 				Rules.updateAllRules();
161 			}
162 		});
163 }

252 		})

255 		});

260 /**
261 Get a metacourse box for the given rule object
262 @param rule a rule object, as given by one of the classes in Rules
263 @param savestr additional rule information associated with a save. See Store.serialize
264 */
265 Box.getMetaBox = function(/**Rule*/ rule, /**String*/ savestr) {
266 	//	title: the title to display
267 	//	course: should contain: 
268 	//		.Credits the number of credits for the metabox
269 	//		.Code a unique code for the metabox
270 	var course = rule.getCourse(savestr);
271 	var title = rule.getTitle();
272 	var ncred = Schedule.getCredits(course);
273 	var dv = $("<div>")
274 		.data("numcredits",ncred)
275 		.data("course",course)
276 		.data("rule",rule)
277 		.data("savestr", savestr)
278 		.data("clone",
279 			(function(rule, savestr) {
280 				return function() {
281 					return Box.getMetaBox(rule, savestr);
282 				};
283 			})(rule, savestr)
284 		)
285 		.addClass("course")
286 		.addClass("metac")
287 		.css({
288 			"width":(Ticks.credw*ncred) + "px",
289 			"border-color":(course.Color) ? Utils.colorstr(course.Color) : "#CCCCFF"
290 		})
291 		.append(
292 			$("<div>")
293 			.addClass("metactitle")
294 			.css({
295 				"background-color":(course.Color) ? Utils.colorstr(course.Color) : "#CCCCFF"
296 			})
297 			.text(title)
298 		)
299 		.mouseover(function() {
300 			var course = $(this).data("course");
301 			var rule = $(this).data("rule");	
302 			
303 			//defer to the rule for the popup contents
304 			var content = rule.getPopup();
305 			
306 			Tips.show($(this),content,300);
307 		})
308 		.mouseout(function() {
309 			Tips.hide();
310 		})
311 		.click(function() {
312 			//fill the search box with the searcher for this metacourse
313 			var searchtext = $(this).data("rule").getSearchText();
314 			
315 			if (searchtext != "") {
316 				$("#search").attr("value",searchtext);
317 				$("#search").focus();
318 				$("#search").select();
319 				Buttons.dosearch();
320 			}
321 		});
322 	Box.makeDraggable(dv);
323 	return dv;
324 }

326 /**
327 Get the source div from a drag event
328 @param e a drag event object
329 */
330 Box.srcFromDrag = function(/**Event*/ e) {
331 	var src = e.originalEvent.currentTarget;//e.originalTarget;
332 	if (!src) {
333 		src = e.originalEvent.srcElement;
334 	}
335 
336 	while (src && (!$(src).data || !$(src).data("numcredits"))) {
337 		src = src.parentNode;
338 	}
339 	return src;
