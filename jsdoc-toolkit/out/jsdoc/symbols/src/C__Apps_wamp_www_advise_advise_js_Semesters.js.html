1 /**
  2 @namespace Singleton used for maintaining the semester drop-down boxen on the left portion of the course manager.
  3 Semesters are indexed starting from 0 as Fall 2000
  4 */
  5 var Semesters = {
  6 };
  7 
  8 /**
  9 the number of semester options to show in the future
 10 */
 11 Semesters.INFUTURE = 4; 
 12 
 13 /**
 14 @param idx the index of the semester to look up
 15 @return "Fall" if the semester is a fall semester, "Spring" otherwise
 16 */
 17 Semesters.fallspring = function(/**int*/ idx) {
 18 	return (idx % 2 == 0) ? "Fall" : "Spring";
 19 };
 20 
 21 /**
 22 @param idx the index of the semester to look up
 23 @return The year of the given semester (2000, 2001, ...)
 24 */
 25 Semesters.year = function(/**int*/ idx) {
 26 	return 2000 + Math.floor((idx+1) / 2);
 27 };
 28 
 29 /**
 30 @param idx the index of the semester to look up
 31 @return The "short year" of the given semester ("'01", "'02", ...)
 32 */
 33 Semesters.yearshort = function(/**int*/ idx) {
 34 	var n = Math.floor((idx+1) / 2);
 35 	if (n < 10) {
 36 		return "'0" + n;
 37 	} else {
 38 		return "'" + n;
 39 	}
 40 };
 41 
 42 /**
 43 @param idx the index of the semester to look up
 44 @return The fall/spring and short year of the semester combined.
 45 */
 46 Semesters.fullstr = function(/**int*/ idx) {
 47 	return Semesters.fallspring(idx) + " " + Semesters.yearshort(idx);
 48 };
 49 
 50 /**
 51 Update all the semester drop-down boxes
 52 */
 53 Semesters.updateselectors = function() {
 54 	for (var x = 0; x < Ticks.NUMSEMS; ++x) {
 55 		Semesters.updateselector(x);
 56 		Semesters.updatealerts(x);
 57 	}
 58 };
 59 
 60 /**
 61 Update the alert boxes for all course boxes in the given semester 
 62 @param sem the semester to update
 63 */
 64 Semesters.updatealerts = function(/**int*/ sem) {
 65 	Schedule.getCourseDivs(sem).each(function(i, el) {
 66 		Tips.updateAlerts($(el));
 67 	});
 68 };
 69 
 70 /**
 71 Update the given semester drop-down selector
 72 @param idx the index of the drop-down box to update (0 to the number of semesters shown in the UI)
 73 */
 74 Semesters.updateselector = function(/**int*/ idx) { //update one semester selector
 75 	//find the correct span to place in the options boxen
 76 	var low = 0;
 77 	var high = 0;
 78 	if (idx == 0) {
 79 		low = curuser.getSemesters()[idx] - Semesters.INFUTURE;
 80 		high = curuser.getSemesters()[idx] + Semesters.INFUTURE;
 81 	} else {
 82 		low = curuser.getSemesters()[idx-1]+1;
 83 		high = curuser.getSemesters()[idx] + Semesters.INFUTURE;
 84 	}
 85 	
 86 	var el = $("#sdesc" + idx).html("");
 87 	for (var x = low; x <= high; ++x) {
 88 		el.append(
 89 			$("<option>")
 90 				.attr("value",x)
 91 				.attr("selected", (x == curuser.getSemesters()[idx]) ? "selected" : "")
 92 				.html(Semesters.fullstr(x))
 93 		);
 94 	}
 95 	
 96 	//HACK: IE makes the boxes explode if we don't reinsert them
 97 	//	after repopulation.
 98 	var elin = el.get(0);
 99 	var pn = elin.parentNode;
100 	pn.removeChild(elin);
101 	pn.appendChild(elin);
102 };
103 
104 /**
105 Triggered by a dropdown select event
106 */
107 Semesters.eventchange = function() { 
108 	var idx = parseInt(this.id.substring(5), 10);
109 	var val = parseInt($(this).attr("value"), 10);
110 	var prevsems = curuser.getSemesters();
111 	for (var x = idx; x < Ticks.NUMSEMS; ++x) {
112 		prevsems[x] = x + (val-idx);
113 	}
114 	
115 	curuser.setSemesters(prevsems);
116 	
117 	for (var x = idx; x < Ticks.NUMSEMS; ++x) {
118 		Semesters.updateselector(x);
119 		Semesters.updatealerts(x);			
120 	}
121 };

123 /**
124 @return the semester value for the given semester index
125 @param idx the semester index (bounded by 0 and NUMSEMS)
126 */
127 Semesters.semesterAt = function(/**int*/ idx) {
128 	return curuser.getSemesters()[idx];
129 };

131 /**
132 @return return the index corresponding to the given semester and year
133 @param fallspring either "Fall" or "Spring"
134 @param year the year, starting from 2000 (eg, 2010 is 10)
135 */
136 Semesters.idxForSem = function(/**String*/ fallspring, /**int*/ year) {
137 	return year*2 + ((fallspring.toLowerCase() == "spring") ? 1 : 0);
138 };
